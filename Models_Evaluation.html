                                   "newbie","channel")],TRUE)
raw_tree <- create_node(train,0,100,treatment_list,response,'control',test_list)

p_idx <- createDataPartition(y = train[ , response], p=0.3, list = FALSE)

pred <- predict.dt.as.df(pruned_tree, test)
